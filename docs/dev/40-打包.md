```css
/* renderer/styles/index */

@import "tailwindcss";

* {
  user-select: none;

  height: 100%;
  width: 100%;

.no-drag {
  -webkit-app-region: no-drag;

  -webkit-app-region: drag;

.dialog-overlay {
  opacity: 0;
  transition: opacity 0.8s ease;
  position: fixed;
  top: 0;
  left: 0;
  bottom: 0;

  width: 10px; /* 滚动条宽度 */

  background: transparent; /* 滚动条轨道透明 */
}

::-webkit-scrollbar-thumb {
  background-color: var(--text-secondary); /* 滚动条滑块颜色 - 中性灰色 */
  border-radius: 4px; /* 圆角 */
  border: 2px solid transparent; /* 边框透明，用于创建间隔效果 */
  background-clip: content-box; /* 重要：确保背景只绘制在内容区域 */

  background-color: #6b7280; /* 悬停时变亮 */

/* 隐藏滚动条上下箭头 */

  display: none;
}

/* 适用于Firefox */
* {
  scrollbar-width: thin; /* 细滚动条 */
  scrollbar-color: var(--text-secondary) transparent; /* 滑块颜色和轨道颜色 */
}
```
### 图标预加载
```ts
// renderer/utils/icons.ts

import { loadIcon } from '@iconify/vue';

export async function preloadIcons() {
  await Promise.all([
    loadIcon('material-symbols:keep-rounded'),
    loadIcon('material-symbols:close'),
    loadIcon('material-symbols:chrome-maximize-outline-sharp'),
    loadIcon('material-symbols:chrome-restore-outline-sharp'),
    loadIcon('material-symbols:chrome-minimize-sharp'),
    loadIcon('material-symbols:chat-outline'),
    loadIcon('material-symbols:settings-outline'),
    loadIcon('material-symbols:light-mode-outline'),
    loadIcon('material-symbols:dark-mode-outline'),
    loadIcon('material-symbols:auto-awesome-outline'),
    loadIcon('material-symbols:arrow-upward')
  ])
}
```
然后用到主入口文件中：
```ts
// renderer/index

// icon预加载
preloadIcons().finally(() => {
  logger.info('icons preloaded');

  createApp(App)
  .use(pinia)
  .use(router)
  .use(components)
  .use(i18n)
  .use(errorHandler)
  .mount('#app');
})

```


## 打包
设置打包后的 exe 名称，软件/安装包图标，优化体验的配置：
```ts
// forge.config.ts

 packagerConfig: {
    name: 'Cecilia',
    icon: '/public/logo.ico',
    // 防止用户篡改源码，优化性能（丝滑畅快）
    asar: true,
  },
  // ...
  config: {
        setupIcon: './public/logo.ico',
        appIcon: './public/logo.ico' 
      },
```
